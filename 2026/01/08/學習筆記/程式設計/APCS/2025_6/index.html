# 2025/6

k=int(input())
x1,y1=map(int,input().split())
x2,y2=map(int,input().split())
p=0
while k>0:
    p+=k
if p%x1==0:
        k-=y1
if p%x2==0:
        k-=y2
print(p)
m,n,k=map(int,input().split())
l=[input()for i in range(m)]
t=[]

ans=0
for values in [list(map(int, input().split()))for _ in range(k)]:

    for i in range(m):
        s = l[i]
        p = values[i] % n
        s = s[-p:] + s[:-p]
        l[i] = s
    
    for i in zip(*l):
        ans+=max(i.count(j)for j in i)
print(ans)
n, t = map(int, input().split())
st = []
ans = 0
l = list(map(int, input().split()))
for i in range(n):
    x = l[i]
    while st and st[-1] <= x:
        ans += st[-1]
        x += st[-1]
        st.pop()
    if x <= t:
        st.append(x)
for x in st:
    ans += x
print(ans)
