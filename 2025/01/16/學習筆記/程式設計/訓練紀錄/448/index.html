# 448

日期: 2025年1月16日

class Solution:
    def findDisappearedNumbers(self, nums: List[int]) -> List[int]:
        l = [False] * int(len(nums)+1)
        for n in nums:
            l[n] = True
        lst = []
        for i in range(1, len(l)):
            if not l[i]:
                lst.append(i)
        return lst
